for line in input():
  chars += line

lookup2 = "ABCDEFGHIJKLMNOPQRSTabcdefghijklmnopqrst"

out = ""

  cur = lookup1.index(char)
  out += lookup2[(cur - prev) % 40]
  prev = cur
